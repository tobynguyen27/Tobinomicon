## Overview

The multiblock projector allows players to visualize complex structures in-game using "ghost" blocks.

## Custom multiblock

You can extend this feature by adding your own custom structure patterns via a standard Minecraft data pack.

<Callout title="Note" type="info">
Astral Generators auto format JSON file name into readable name.  
For example: `fusion_reactor.json` to `Fusion Reactor`
</Callout>

To register a new multiblock pattern, you must place your JSON file in `data/astralgenerators/multiblocks/<your_structure_name>.json`

<Callout title="Attention" type="warning">
    You must place yor JSON file in `astralgenerators` namespace, otherwise it will be ignored.  
    **Ignored**: `data/testmod/multiblocks/reactor.json`  
    **Accepted**:`data/astralgenerators/multiblocks/reactor.json`
</Callout>

<auto-type-table title="MultiblockDefinition" type='export interface MultiblockDefinition {
    /**
     * The offset coordinates relative to the projector block.
     * This determines where the structure starts rendering.
     */
    origin: {
        x: number;
        y: number;
        z: number;
    };

    /**
     * A dictionary mapping single characters (used in the pattern) to block data.
     * Keys must be single characters (e.g., "a", "b", "m").
     */
    keys: Record<string, { block: string }>;

    /**
     * A 3D representation of the structure layers.
     * - Outer Array: Layers (Y-axis, from bottom to top).
     * - Inner Array: Rows (Z-axis).
     * - String: Columns (X-axis) composed of character keys.
     */
    pattern: string[][];
}'></auto-type-table>

```json title="boiler.json"
{
    "origin": {
        "x": -1,
        "y": 0,
        "z": 2
    },
    "keys": {
        "a": {
            "block": "astralgenerators:boiler_casing"
        },
        "b": {
            "block": "astralgenerators:firebox_casing"
        },
        "m": {
            "block": "astralgenerators:boiler_controller"
        }
    },
    "pattern": [
        ["bbb", "bbb", "bbb"],
        ["ama", "a a", "aaa"],
        ["aaa", "a a", "aaa"],
        ["aaa", "aaa", "aaa"]
    ]
}
```
